CREATE VIEW [dbo].[VIEW_MEMBERBALANCE_WITH_HOLD]
AS
SELECT 
    mm.Msrno,
    mm.LoginID,
    mm.Name,
    mm.Mobile,
    mm.Email,
    mm.Address,
    mm.RoleID,
    mm.ParentID,

    ew.Balance AS WalletBalance,
    aew.Balance AS AEPSBalance,
    pew.Balance AS ProfitBalance,

    mm.HoldAmt AS HoldBalance,

    -- Usable balances after deducting HoldAmt
    (ew.Balance - ISNULL(mm.HoldAmt,0)) AS UsableBalance,
    (aew.Balance - ISNULL(mm.HoldAmt,0)) AS UsableAEPSBalance
FROM dbo.Member mm
INNER JOIN dbo.TBL_EWALLETBALANCE ew 
    ON ew.Msrno = mm.Msrno
INNER JOIN dbo.tblMLM_AEPS2EWalletBalance aew 
    ON aew.Msrno = mm.Msrno
INNER JOIN dbo.PROFITBalance pew 
    ON pew.Msrno = mm.Msrno;
GO
